public class Solution {
  public int[] rainbowSort(int[] array) {
    if (array == null || array.length == 0) return array;

    int left = 0; // 左边界指针
    int right = array.length - 1; // 右边界指针
    int i = 0; // 当前遍历指针

    while (i <= right) {
      if (array[i] == -1) {
        swap(i, left, array);
        left++;
        i++; // 左边界增加后，指针i也需要增加
      } else if (array[i] == 1) {
        swap(i, right, array);
        right--;
        // 不需要增加指针i，因为交换后的元素需要重新检查
      } else {
        i++; // 跳过值为0的元素
      }
    }

    return array;
  }

  public void swap(int i, int j, int[] array) {
    int temp = array[i];
    array[i] = array[j];
    array[j] = temp;
  }
